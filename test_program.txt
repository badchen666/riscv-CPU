## RISC-V 32I 测试程序 — 累加 1+2+...+10 = 55
##
## 寄存器约定:
##   x1  = 11         (循环终止条件, i > 10 时 i==11)
##   x2  = 累加器 sum (最终 = 55)
##   x5  = 1          (步长)
##   x6  = 循环变量 i (1..10)
##   x3  = 从内存读回的 sum (验证 load/store)
##   x4  = x3 的副本  (验证 MEM-EX 前递)
##   x7  = 0xDEAD     (结束标志)
##
## 测试覆盖:
##   - I-type  : addi
##   - R-type  : add
##   - Store   : sw
##   - Load    : lw  (含 Load-Use 冒险)
##   - Branch  : bne (含控制冒险)
##   - JAL     : 原地死循环
##   - 前递    : EX-EX (add 连续使用前一条结果)
##   - 前递    : MEM-EX (lw 结果被后续指令使用)
##
## ============================================================
## 地址  机器码      汇编
## ============================================================
## 0x00  00B00093   addi  x1,  x0, 11   # x1 = 11 (哨兵)
## 0x04  00000113   addi  x2,  x0,  0   # x2 = 0  (sum)
## 0x08  00100293   addi  x5,  x0,  1   # x5 = 1  (步长)
## 0x0C  00100313   addi  x6,  x0,  1   # x6 = 1  (i=1)
##
## loop: (PC=0x10)
## 0x10  00610133   add   x2,  x2, x6   # sum += i  [EX-EX前递]
## 0x14  00530313   addi  x6,  x6,  1   # i++       [EX-EX前递: x6]
## 0x18  FE6096E3   bne   x1,  x6, loop # if x1!=x6 goto loop  (偏移=-20)
##                                        # 循环结束时 x2=1+2+...+10=55, x6=11
##
## 0x1C  00202023   sw    x2,  0(x0)    # mem[0] = 55
## 0x20  00002183   lw    x3,  0(x0)    # x3 = mem[0] = 55  [Load-Use冒险]
## 0x24  00018233   add   x4,  x3, x0   # x4 = x3  [MEM-EX前递]
## 0x28  0DEAD3B7   lui   x7,  0xDEAD   # x7 = 0xDEAD000 (结束标志)
## 0x2C  0000006F   jal   x0,  0        # 原地死循环
##
## 期望结果:
##   x1  = 11
##   x2  = 55
##   x3  = 55
##   x4  = 55
##   x5  = 1
##   x6  = 11
##   x7  = 0x0DEAD000
##   mem[0..3] = 0x00000037 (55 = 0x37)
